package com.senzzzi.sword.validator;

import com.senzzzi.sword.model.persistence.TaskEntity;
import com.senzzzi.sword.repository.TaskRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

import java.util.Optional;

@Component
@RequiredArgsConstructor
public class TaskPerformerValidator {

    private final TaskRepository taskRepository;

    public boolean isMyOwnTask(Long taskId, String username) {
        Optional<TaskEntity> taskEntity = taskRepository.findById(taskId);
        if (taskEntity.isEmpty()) {
            return false;
        }

        if (taskEntity.get().getCreatedBy().equals(username)) {
            return true;
        }

        return false;
    }
}
